Hooks в React

Hooks — это функции, которые позволяют использовать состояние и другие возможности React в функциональных компонентах.
Основные хуки: useState , useEffect , useRef , useContext , useReducer , useCallback , useMemo .

useState — состояние компонента


function Counter() {
  const [count, setCount] = React.useState(0);
  return (
    <button onClick={() => setCount(count + 1)>{count}</button>
  );
}
useState возвращает массив: [текущее значение, функция для обновления].

useEffect — побочные эффекты


function Timer() {
  const [time, setTime] = React.useState(0);
  React.useEffect(() => {
    const id = setInterval(() => setTime(t => t + 1), 1000);
    return () => clearInterval(id);
  }, []);
  return <div>Время: {time}</div>;
}
useEffect запускает код после рендера. Массив зависимостей — когда перезапускать.

Кастомные хуки


function useOnlineStatus() {
  const [online, setOnline] = React.useState(navigator.onLine);
  React.useEffect(() => {
    function handleStatus() { setOnline(navigator.onLine); }
    window.addEventListener('online', handleStatus);
    window.addEventListener('offline', handleStatus);
    return () => {
      window.removeEventListener('online', handleStatus);
      window.removeEventListener('offline', handleStatus);
    };
  }, []);
  return online;
}
function Status() {
  const online = useOnlineStatus();
  return <div>{online ? 'Онлайн' : 'Оффлайн'}</div>;
}
Кастомные хуки — простые функции, начинающиеся с "use", для переиспользования логики.

Практика: счетчик и таймер


<div id="root"></div>
<script type="text/babel">
function Counter() {
  const [count, setCount] = React.useState(0);
  return (
    <button onClick={() => setCount(count + 1)}>{count}</button>
  );
}
function Timer() {
  const [time, setTime] = React.useState(0);
  React.useEffect(() => {
    const id = setInterval(() => setTime(t => t + 1), 1000);
    return () => clearInterval(id);
  }, []);
  return <div>Время: {time}</div>;
}
function App() {
  return (
    <div>
      <Counter />
      <Timer />
    </div>
  );
}
const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
</script>

Советы и типичные ошибки

  • Все хуки вызываются только на верхнем уровне компонента!
  • Не вызывайте хуки внутри условий, циклов или вложенных функций.
  • useEffect с пустым массивом зависимостей выполняется один раз (аналог componentDidMount).
  • Переиспользуйте логику через кастомные хуки.

Лучшие практики

  • Всегда начинайте кастомные хуки с "use".
  • Не дублируйте логику — выносите её в хуки.
  • Документируйте хуки (PropTypes, JSDoc).
  • Для сложных состояний используйте useReducer.

Примеры hook-логики

useRef — ссылка на DOM-элемент


function FocusInput() {
  const inputRef = React.useRef();
  function focus() {
    inputRef.current.focus();
  }
  return (
    <div>
      <input ref={inputRef} />
      <button onClick={focus}>Фокус</button>
    </div>
  );
}

useMemo — оптимизация вычислений


function Expensive({n}) {
  const result = React.useMemo(() => compute(n), [n]);
  return <div>Результат: {result}</div>;
}

useCallback — мемоизация функций


const handleClick = React.useCallback(() => alert('Клик!'), []);
